#include<stdio.h>
struct Employee
{
    int id;
    float salary;
};
int main()
{
    struct Employee emp[3];
    int i;
    for(i = 0;i < 3; i++)
    {
        printf("Enter details of employee %d\n",i + 1);
        printf("Employee ID: ");
        scanf("%d", &emp[i].id);
        printf("Salary:");
        scanf("%d",&emp[i].salary);
        printf("\n");
    }
    
    printf("Employee Details:\n");
    for(i = 0; i < 3; i++)
    {
        printf("Employee %d\n", i + 1);
        printf("ID: %d\n",emp[i].id);
        printf("Salary: %.2f\n, emp[i}.salary");
        printf("\n");
        
    }
    return 0;
}
