#include<stdio.h>
#include<stdlib.h>
#include<string.h>

struct Node{
    char station[10];
    struct Node *prev;
    struct Node *next;
    
};
struct Node* insertEnd(struct Node *head, char station[]) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    strcpy(newNode->station, station);
    newNode->prev = NULL;
    newNode->next = NULL;
    
    if(head == NULL) {
        return newNode;
        
    }
    
    struct Node *temp = head;
    while (temp->next != NULL) {
        temp = temp->next;
    }
    temp->next = newNode;
    newNode->prev = temp;
    
    return head;
}
void display(struct Node *head){
    struct Node *temp = head;
    while(temp != NULL){
        printf("%s <-> ",temp->station);
        temp = temp->next;
    }
    printf("NULL\n");
}

int main(){
    struct Node *head = NULL;
    
    head = insertEnd(head, "A1");
    head = insertEnd(head, "B2");
    head = insertEnd(head, "C3");
    
    display(head);
    return 0;
}
