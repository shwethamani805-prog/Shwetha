#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Product {
    char name[50];
    int price;
};

struct Order {
    int orderId;
    struct Product product;
    char status[20];
    struct Order *next;
};

int main() {
    struct Order *head = NULL, *second = NULL;

    
    head = (struct Order *)malloc(sizeof(struct Order));
    head->orderId = 201;
    strcpy(head->product.name, "Mouse");
    head->product.price = 500;
    strcpy(head->status, "Delivered");

    
    second = (struct Order *)malloc(sizeof(struct Order));
    second->orderId = 202;
    strcpy(second->product.name, "Keyboard");
    second->product.price = 1500;
    strcpy(second->status, "Pending");

    
    head->next = second;
    second->next = NULL;

    
    struct Order *temp = head;
    while (temp != NULL) {
        printf("%d %s %d %s -> ",temp->orderId,  temp->product.name,  temp->product.price,temp->status);
        temp = temp->next;
    }
    printf("NULL");

    return 0;
}
